3. Perfect number Check

      
var Perfect_Check = (N) => 
{   let sum = 0;
    
    for(let i = 1; i < N; i++) {
      if(N%i === 0) {
       sum+=i;
    }
    }
    
    if(N === sum) {
      return "YES";
    }
    
    else {
      return "NO";
    }
      
        
};
